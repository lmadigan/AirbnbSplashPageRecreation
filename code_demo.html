$(".video-1").mousemove( hoverVideo );
$(".video-1").mouseout( hideVideo );

function hoverVideo(e) {
e.currentTarget.play();
e.muted = false;
}

function hideVideo(e) {
e.currentTarget.pause();
e.currentTarget.currentTime = 0;
}
